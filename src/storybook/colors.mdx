import { colorGroups } from "@/storybook/colorGroup";
import { BRAND_COLORS } from "@/styles/brandColor";
import { ColorItem, ColorPalette, Meta } from "@storybook/addon-docs";

<Meta title="System/Colors" />

# Color Palette

<ColorPalette>
  {colorGroups.map(({ title, subtitle, key, colorSteps }) => (
    <ColorItem
      key={key}
      title={title}
      subtitle={subtitle}
      colors={
        colorSteps && colorSteps.length > 0
          ? colorSteps.reduce((acc, step) => {
              acc[step] =
                BRAND_COLORS[`${key}_${step}`] !== undefined
                  ? BRAND_COLORS[`${key}_${step}`]
                  : step === "DEFAULT"
                  ? BRAND_COLORS[key]
                  : null;
              return acc;
            }, {})
          : { DEFAULT: BRAND_COLORS[key] } // colorSteps가 없을 경우 기본 색상만 사용
      }
    />
  ))}
</ColorPalette>
